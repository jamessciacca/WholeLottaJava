//working on creating files using the FS module tool
//-- first step is to create a variable to hold the fs module data
const fs = require('fs');

//writeFile function with filename, content, and callback function
fs.writeFile('newFile.txt', 'Learn Node FS module', function(err){
    if(err)throw err;
    console.log('File is created successfully.');
});

//the call back function lets the user know if an error occurs

fs.writeFile('index.html', `<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <h1>Hello</h1>
</body>
</html>`, function(err){
    if(err)throw err;
    console.log('File is created successfully.');
});
